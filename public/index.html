FieldKit Demo

TextField

TextField is the base class that wraps an <input type="text"> with helpers that give you advanced control over the field. 

<input type="text" id="text-field-example" />
var field = new FieldKit.TextField(document.getElementById('text-field-example'));
field.setFocusedPlaceholder('John Smith');
field.setUnfocusedPlaceholder('First Name');

NumberFormatter

The NumberFormatter provides formatting for all kinds of numbers, percentages, currencies, etc. 


            

          

        


        
<input type="text" id="number">

        
var field = new FieldKit.TextField(document.getElementById('number'));
field.setFormatter(new FieldKit.NumberFormatter()
  .setNumberStyle(Math.random() > 0.5 ?
    FieldKit.NumberFormatter.Style.PERCENT :
    FieldKit.NumberFormatter.Style.CURRENCY));
field.setValue((Math.random() > 0.5 ? 1 : -1) * Math.random()*10);

      

    



    

      

        
ExpiryDateField / ExpiryDateFormatter

      

    


    

      

        

          The ExpiryDateField wraps
          ExpiryDateFormatter for credit card expiry
          dates, preventing the input of nonsense months or non-date
          characters.  It also adds the preceding zero and the slash between
          the month and year automatically, but treats them as if they are not
          there. In order to do this predictably it forces the cursor to remain
          at the end of the field and prevents any kind of selection.
        


        

          

            


            

              
“4” → “04/”
“1212” → “12/12”
“12/” + backspace → “1”

            

          

        


        
<input type="text" id="expiry" placeholder="mm/yy">

        
new FieldKit.ExpiryDateField(document.getElementById('expiry'));

      

    



    

      

        
CardTextField / AdaptiveCardFormatter

      

    


    

      

        

          The CardTextField and AdaptiveCardFormatter
          provide formatting for credit card PANs, preventing the input of
          non-digit characters and automatically breaking into digit groups
          according to the rules for the entered card type.
        


        

          

            


            

              
“411111111” → “4111 1111 1”  # visa
“3725123456” → “3725 123456” # amex
“4111 ” + backspace → “411”

            

          

        


        
<input type="text" id="card-number" placeholder="1234 5678 9012 3456">
<div id="card-type">Card type: <span id="card-type-span">unknown</span></div>

        
var field = new FieldKit.CardTextField(document.getElementById('card-number'));
field.setCardMaskStrategy(FieldKit.CardTextField.CardMaskStrategy.DoneEditing);
field.setDelegate({
  textDidChange: function() {
    document.getElementById('card-type-span').innerHTML = (field.cardType() || 'unknown');
  }
});

      

    



    

      

        
SocialSecurityNumberFormatter

Writing a custom formatter

      

    


    

      

        

          Formatters are objects that implement the Formatter interface:
        


        
interface Formatter
{
  string format(any objectValue);
  any parse(string stringValue);

  [Optional] boolean isChangeValid(Change change);
}

interface Change
{
  attribute TextField field;
  attribute TextState current;
  attribute TextState proposed;
  attribute TextChange inserted;
  attribute TextChange deleted;
}

interface TextState
{
  attribute string text;
  attribute Range selectedRange;
}

interface TextChange
{
  attribute string text;
  attribute unsigned long start;
  attribute unsigned long end;
}

interface Range
{
  attribute unsigned long start;
  attribute unsigned long length;
}



        
Character Replacement


        

          This example replaces any ‘b’ it finds with ‘a’.
          It does this using all three of the methods on Formatter.
        


        

          

            


            

              
“Hot tub” → “Hot tua”
“But why?” → “Aut why?”

            

          

        


        
<input type="text" id="b-to-a" placeholder="changes 'b' to 'a'">

        
var field = new FieldKit.TextField(document.getElementById('b-to-a'));
field.setFormatter({
  format: function(value){ return value.replace(/b/g, 'a').replace(/B/g, 'A'); },
  parse: function(value){ return value; },

  isChangeValid: function(change) {
    change.proposed.text = this.format(this.parse(change.proposed.text));
    return true;
  }
});

      

    


    

      

        
Controlling the Selection


        

          Formatters not only have control over the editable text but also the
          selection, if they so choose. Formatters can exercise this control by
          implementing the #isChangeValid(change) method and
          changing the proposed selection. The example below does not impose
          any restrictions on what can be typed or how to display the value,
          but disables all selection. Because of browser limitations you can
          still select text using the mouse and by tabbing into this field.
        


        

          

            

          

        


        
<input type="text" id="no-selection" placeholder="no selection">

        
var field = new FieldKit.TextField(document.getElementById('no-selection'));
field.setFormatter({
  format: function(value){ return value; },
  parse: function(value){ return value; },

  isChangeValid: function(change) {
    var range = change.proposed.selectedRange;

    // Check to see whether the user tried to select something.
    if (range.length !== 0) {
      if (change.field.selectionAnchor() === range.start) {
        // Move to the end since the selection extended
        // from the beginning toward the end (e.g. shift+right).
        range.start += range.length;
      }

      range.length = 0;
    }

    return true;
  }
});

      

    


    

      

        
Form Interaction


        

          FieldKit text fields should behave essentially as expected with
          regard to forms. Tabbing from one field to the next should work.
          Pressing ENTER should submit the form. Etc.
        


        

          

            

          

        


        
<form id="form-test">
  <label for="name">Name</label> <input type="text" id="name" name="name" class="form-test" placeholder="name">
  <label for="email">Email</label> <input type="text" id="email" name="email" class="form-test" placeholder="email">
  <label for="phone">Phone</label> <input type="text" id="phone" name="phone" class="form-test" placeholder="phone">
  <input type="submit" class="btn btn-primary">
</form>

        
var fields = Array.prototype.filter.call(document.getElementsByClassName('form-test'), function(el) {
  if(el.id === 'phone') return new FieldKit.TextField(el, new FieldKit.PhoneFormatter());
  return new FieldKit.TextField(el);
});
var form = document.getElementById('form-test'),
    popup = function(event) {
      event&&event.preventDefault();
      alert('Hello, ' + fields[0].value + ' ' + fields[1].value);
      return false;
    };

if (form.addEventListener) {
  form.addEventListener('submit', popup, false);  //Modern browsers
} else if (form.attachEvent) {
  form.attachEvent('onsubmit', popup);            //Old IE
}
